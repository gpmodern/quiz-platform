package com.quizplatform.dto;

public class QuizDto {
    private Long id;
    private String title;

    public QuizDto() {}

    public QuizDto(Long id, String title) {
        this.id = id;
        this.title = title;
    }

    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }
}
